from collections import Counter

def count_good_persons(S1, S2):
    
    freq1 = Counter(S1)
    freq2 = Counter(S2)
    
    
    good_person_count = 0
    for char in freq1:
        if freq1[char] == 2 and freq2.get(char, 0) == 3:
            good_person_count += 1
    return good_person_count

S1 = "aabbjfd"
S2 = "aaabbbjkg"
print(count_good_persons(S1, S2))
